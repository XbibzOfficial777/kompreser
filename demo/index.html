Code Examples

See how easy it is to integrate Kompreser

Basic Usage 

import Kompreser from '@xbibzlibrary/kompreser';

// Initialize
const kompreser = new Kompreser({
  quality: 0.8,
  format: 'auto',
  progressive: true
});

// Compress a single image
const result = await kompreser.compress(file);
console.log(`Reduced by ${Math.round(result.metadata.compressionRatio * 100)}%`);

Batch Processing 

// Compress multiple images
const results = await kompreser.compressBatch(files, {
  quality: 0.85,
  format: 'webp'
});

results.forEach((result, index) => {
  if (result.success) {
    console.log(`Image ${index}: ${result.compressionRatio}% reduction`);
  }
});

Advanced Configuration 

const kompreser = new Kompreser({
  // Compression
  quality: 0.9,
  format: 'avif',
  progressive: true,
  
  // Performance
  useWorkers: true,
  maxWorkers: 8,
  enableWebAssembly: true,
  
  // Error handling
  enableRecovery: true,
  maxRetries: 3,
  
  // Security
  maxFileSize: 50 * 1024 * 1024,
  sanitizeMetadata: true
});
